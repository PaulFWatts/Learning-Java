import java.util.Scanner;

/**
 * Demonstrates the use of conditionals to control whether debugging messages
 * are displayed or not. The actual program is an automated letter writer.
 */
public class DebugExperiment {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String sender, recipient, gift;
        char allOK;
        final boolean DEBUGON = false;
        final char YES = 'y';

        if (DEBUGON) {
            System.out.println("program beginning"); // debugging statement
        }

        System.out.println("This program will automatically generate thank you letters");
        System.out.print("Enter your name: ");
        sender = sc.nextLine();
        System.out.print("Enter the name of the person who gave the gift: ");
        recipient = sc.nextLine();
        System.out.print("What was the gift? ");
        gift = sc.nextLine();
        System.out.println("Values entered are:");
        System.out.println("sender: " + sender);
        System.out.println("recipient: " + recipient);
        System.out.println("gift: " + gift);

        System.out.println("Enter y if these are all correct");
        allOK = sc.next().charAt(0);

        if (allOK == YES) {
            if (DEBUGON) {
                System.out.println("condition evaluated to true - continuing with letter"); // debugging statement
            }
            System.out.println("Dear " + recipient + ",");
            System.out.println("\t Thank you very much for the beautiful " + gift + ".");
            System.out.println("I will enjoy using it every day. ");
            System.out.println("I hope you are well and saving up for Christmas." );
            System.out.println("love from");
            System.out.println(sender);
        } else {
            if (DEBUGON) {
                System.out.println("condition evaluated to false - no letter");// debugging statement
            }
        }
        if (DEBUGON) {
            System.out.println("Program over" );// debugging statement
        }
    }
}
