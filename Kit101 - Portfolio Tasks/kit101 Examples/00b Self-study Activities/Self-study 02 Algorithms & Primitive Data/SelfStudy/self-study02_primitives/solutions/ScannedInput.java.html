/**
    ScannedInput: Trying out interactive input using the Scanner class
    @author 
    @version 
*/

import java.util.Scanner;

public class ScannedInput 
{

    public static void main(String[] args) 
    {
        System.out.println( "program begins" );
        Scanner scan;
        int testInt;
        double testDouble;
        char testChar;
        boolean testBoolean;
        String s;
        
        scan = new Scanner(System.in);
        testInt = 42;
        testDouble = 12.43;
        testChar = '!';
        testBoolean = true;
        
        System.out.println("Please enter an integer");
        testInt = scan.nextInt();
        
        System.out.println("Please enter a double");
        testDouble = scan.nextDouble();
        
        System.out.println("Please enter a char");
        s = scan.next();
        testChar = s.charAt(0);
        
        System.out.println("Please enter a boolean");
        testBoolean = scan.nextBoolean();
        
        System.out.println("testInt value is: " + testInt);
        System.out.println("testDouble value is: " + testDouble);
        System.out.println("testChar value is: " + testChar);
        System.out.println("testBoolean value is: " + testBoolean);
        
        System.out.println("program over");     
    }
}
