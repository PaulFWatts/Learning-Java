public class LastDigit
{

    public static void main(String[] args) 
    {
        System.out.println( "program begins" );
        
        //int storage for originalNumber
        //int storage for lastDigit
        int originalNumber;
        int lastDigit;
        //originalNumber becomes number to divide up (say 123)
        originalNumber = 123;
        //lastDigit becomes originalNumber % 10
        lastDigit = originalNumber % 10;
        //display lastDigit
        System.out.println("Last digit is: " + lastDigit);

        
        System.out.println("program over");     
    }
}
