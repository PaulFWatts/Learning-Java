public class ConvertToInches
{

    public static void main(String[] args) 
    {
        System.out.println( "program begins" );
        
        //constant double storage for CM_PER_INCH becomes 2.54
        final double CM_PER_INCH = 2.54;
        //double storage for cmDist
        //double storage for inDist
        double cmDist;
        double inDist;
        //cmDist becomes distance to convert (say 20)
        cmDist = 20;
        //inDist becomes cmDist /CM_PER_INCH
        inDist = cmDist /CM_PER_INCH;
        //display inDist
        System.out.println("Distance in inches is: " + inDist);
        
        System.out.println("program over");     
    }
}
