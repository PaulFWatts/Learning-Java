public class Swapping
{

    public static void main(String[] args) 
    {
        System.out.println( "program begins" );
        
        //int storage for number1
        //int storage for number1
        //int storage for tempNumber
        int number1, number2, tempNumber;
        //number1 becomes first value to store (say 15)
        number1 = 15;
        //number2 becomes second value to store (say 20)
        number2 = 20;
        //display number1 number2 (to check the original values)
        System.out.println("number 1: " + number1 + " number 2:" + number2 );
        //tempNumber becomes number1
        tempNumber = number1;
        //number1 becomes number2
        number1 = number2;
        //number2 becomes tempNumber
        number2 = tempNumber;
        //display number1 number2 (to check the final values)
        System.out.println("number 1: " + number1 + " number 2:" + number2 );

        
        System.out.println("program over");     
    }
}
